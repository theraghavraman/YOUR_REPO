                e.preventDefault();
                const target = document.querySelector(this.getAttribute('href'));
                if (target) {

        // Scroll Animations
        const observerOptions = {
            threshold: 0.1,
            rootMargin: '0px 0px -100px 0px'
        };
        const observer = new IntersectionObserver(function(entries) {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.classList.add('animated');
                }
            });
        }, observerOptions);
        document.querySelectorAll('.animate-on-scroll').forEach(el => {
            observer.observe(el);
        });

        function changeTitle() {
            if (titleElement) {
                titleElement.style.opacity = '0';
                setTimeout(() => {
                    titleIndex = (titleIndex + 1) % titles.length;
                    titleElement.textContent = titles[titleIndex];
                    titleElement.style.opacity = '1';
                }, 500);
            }
        }

        // Typing Effect for Hero Description
        const heroDescription = document.querySelector('.hero .description');
        if (heroDescription) {
            const originalText = heroDescription.textContent;
            heroDescription.textContent = '';
            function typeText(text, element, speed = 50) {
                let i = 0;
                const timer = setInterval(() => {
                    if (i < text.length) {
                        element.textContent += text.charAt(i);
                        i++;
                    } else {
                        clearInterval(timer);
                    }
                }, speed);
            }
            setTimeout(() => {
                typeText(originalText, heroDescription, 30);
            }, 2000);

                const rect = this.getBoundingClientRect();
                const x = e.clientX - rect.left;
                const y = e.clientY - rect.top;
                const centerX = rect.width / 2;
                const centerY = rect.height / 2;
                const rotateX = (y - centerY) / 10;
                const rotateY = (centerX - x) / 10;
                this.style.transform = `perspective(1000px) rotateX(${rotateX}deg) rotateY(${rotateY}deg) translateZ(10px)`;
            });

          const items = document.querySelectorAll('.cert-item');
          items.forEach(item => {
            const header = item.querySelector('.cert-header');
            const panel = item.querySelector('.cert-panel');
            if (!header || !panel) return;

            panel.style.height = '0px';

            header.addEventListener('click', () => {
              const expanded = header.getAttribute('aria-expanded') === 'true';

              // Single-open behavior
              items.forEach(other => {
                const h = other.querySelector('.cert-header');
                const p = other.querySelector('.cert-panel');
                if (other !== item && h && p) {
                  h.setAttribute('aria-expanded', 'false');
                  p.style.height = '0px';
                }
              });

              header.setAttribute('aria-expanded', String(!expanded));
              if (!expanded) {
                panel.style.height = panel.scrollHeight + 'px';
              } else {
                panel.style.height = '0px';
              }
            });

            window.addEventListener('resize', () => {
              if (header.getAttribute('aria-expanded') === 'true') {
                panel.style.height = panel.scrollHeight + 'px';
              }
            });
          });
        })();
